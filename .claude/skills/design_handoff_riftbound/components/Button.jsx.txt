import React from 'react';

const VARIANTS = {
  primary:   { background: 'var(--gold)', color: 'var(--text-on-gold)', border: '1px solid var(--gold)' },
  cyan:      { background: '#00bcff22', color: 'var(--cyan)', border: '1px solid var(--cyan)', boxShadow: 'var(--glow-cyan)' },
  secondary: { background: 'var(--surface-elevated)', color: 'var(--text-primary)', border: '1px solid var(--border-strong)' },
  danger:    { background: 'var(--loss-soft)', color: 'var(--loss)', border: '1px solid var(--loss)' },
  ghost:     { background: 'transparent', color: 'var(--text-secondary)', border: '1px solid transparent' },
};

const SIZES = {
  sm: { padding: '8px 16px', fontSize: 13 },
  md: { padding: '14px 24px', fontSize: 15 },
  lg: { padding: '18px 28px', fontSize: 17 },
};

/**
 * Primary action button. Gold by default; `cyan` glows like lit mana for the
 * hero/active action; secondary/danger/ghost for the rest.
 */
export function Button({
  label,
  children,
  onClick,
  variant = 'primary',
  size = 'md',
  loading = false,
  disabled = false,
  fullWidth = false,
  icon = null,
  style = {},
}) {
  const off = disabled || loading;
  return (
    <button
      onClick={onClick}
      disabled={off}
      style={{
        borderRadius: 'var(--r-md)',
        fontFamily: 'var(--font-body)',
        fontWeight: 700,
        letterSpacing: '0.02em',
        display: 'inline-flex',
        alignItems: 'center',
        justifyContent: 'center',
        gap: 8,
        cursor: off ? 'not-allowed' : 'pointer',
        opacity: off ? 0.45 : 1,
        transition: 'opacity var(--dur-fast) var(--ease-out), filter var(--dur-fast) var(--ease-out)',
        width: fullWidth ? '100%' : 'auto',
        ...SIZES[size],
        ...VARIANTS[variant],
        ...style,
      }}
    >
      {icon}
      {loading ? '…' : (label ?? children)}
    </button>
  );
}
