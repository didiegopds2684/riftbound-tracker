<script type="text/babel" data-presets="react">
  const { ChampionDetailScreen, ScoreboardScreen, LegendsScreen, MatchesScreen, AuthScreen, ProfileScreen } = window;

  const TABS = [
    { id: 'jogar',    label: 'Jogar',    icon: 'swords' },
    { id: 'partidas', label: 'Partidas', icon: 'clipboard-list' },
    { id: 'legends',  label: 'Legends',  icon: 'trophy' },
    { id: 'perfil',   label: 'Perfil',   icon: 'user' },
  ];

  function StatusBar() {
    return (
      <div className="statusbar">
        <span>21:30</span>
        <span className="dots">
          <i data-lucide="signal"></i>
          <i data-lucide="wifi"></i>
          <i data-lucide="battery-full"></i>
        </span>
      </div>
    );
  }

  function App() {
    const [authed, setAuthed] = React.useState(false);
    const [tab, setTab] = React.useState('legends');
    const [detailId, setDetailId] = React.useState(null);

    React.useEffect(() => { window.lucide && window.lucide.createIcons(); });

    if (!authed) {
      return (
        <div className="phone">
          <StatusBar />
          <div className="screen"><AuthScreen onLogin={() => setAuthed(true)} /></div>
        </div>
      );
    }

    const { LEGEND_STATS, PROFILE, byId } = window.RBData;

    let body;
    if (detailId) {
      const stats = LEGEND_STATS[detailId] ?? LEGEND_STATS['1'];
      body = <ChampionDetailScreen stats={stats} isFavorite={detailId === PROFILE.favorite_legend_id} onBack={() => setDetailId(null)} />;
    } else if (tab === 'jogar') {
      body = <ScoreboardScreen />;
    } else if (tab === 'partidas') {
      body = <MatchesScreen />;
    } else if (tab === 'legends') {
      body = <LegendsScreen onOpen={(id) => setDetailId(LEGEND_STATS[id] ? id : '1')} />;
    } else {
      body = <ProfileScreen />;
    }

    return (
      <div className="phone">
        <StatusBar />
        <div className="screen">{body}</div>
        <div className="tabbar">
          {TABS.map((t) => (
            <button key={t.id} className={'tab' + (!detailId && t.id === tab ? ' active' : '')}
              onClick={() => { setDetailId(null); setTab(t.id); }}>
              <i data-lucide={t.icon}></i>
              <span className="lbl">{t.label}</span>
            </button>
          ))}
        </div>
      </div>
    );
  }

  ReactDOM.createRoot(document.getElementById('root')).render(<App />);
</script>
